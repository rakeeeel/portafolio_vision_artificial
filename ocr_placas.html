import cv2
import pytesseract

pytesseract.pytesseract.tesseract_cmd=r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

placas_autorizadas=['XYZ789','DMB-123','BPB-543','QUE987','BWU876']
placa=cv2.imread("placas.jpg")
placanueva=cv2.resize(placa,(800,300))

placa_gris=cv2.cvtColor(placanueva,cv2.COLOR_BGR2GRAY)

placa_res=cv2.threshold(placa_gris,120,255,cv2.THRESH_BINARY)[1]

texto_placa=pytesseract.image_to_string(placa_res,lang='spa')

texto_limpio=texto_placa.strip().replace(" ","")

if texto_limpio in placas_autorizadas:
    print("ACCESO CONCEDIDO!!,\n Bienvenido al estacionamiento del CBTIS144")
else:
    print("ACCESO DENEGADO!!,\n No esta registrada esa placa")

print("Texto detectado")
print(texto_placa)

cv2.imshow("Imagen analizada", placa_res)
cv2.waitKey(0)
cv2.destroyAllWindows()
