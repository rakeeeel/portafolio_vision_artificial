#C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR
import cv2 #importa pythoncv para manipular
import pytesseract #libreria OCR

pytesseract.pytesseract.tesseract_cmd=r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

#cargar imagen
imgtexto=cv2.imread("texto6.jpg")

#redimencionar el tama√±o de la imagen
imgfinal=cv2.resize(imgtexto,(800,400))

#converir a escala de grises
img_gris=cv2.cvtColor(imgfinal,cv2.COLOR_BGR2GRAY)

#separamos el fondo con el texto
img_res=cv2.threshold(img_gris,128, 255,cv2.THRESH_BINARY)[1]
#devuelve 2 valores: los pixeles y la imagen procesada

#configurar OCR
texto=pytesseract.image_to_string(img_res,lang="spa")
#como parametro le pasamos la variable con la imagen procesada y el idioma
#imprimimos el titulo
print("texto detectado")
print(texto)

cv2.imshow("imagen analizada",imgfinal)
cv2.waitKey(0)
cv2.destroyAllWindows()
