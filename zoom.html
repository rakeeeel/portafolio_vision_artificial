7. Zoom de imagen

üìù C√≥digo del Ejemplo

def zoom():
    #23 de octubre
    import cv2
    #carga la imagen de una variable
    img_original=cv2.imread("Gatooo.jpg")
    #creando variable para el valor del zoom original
    zoom=1.0

    while True:
        #espera 4 parametros: la variable de la imagen original, la cantidad del zoom, controlar
        #el tama√±o en el eje x y en el eje y
        img=cv2.resize(img_original,None,fx=zoom,fy=zoom)
        #espera 2 parametro:titulo de la ventana y la imagen a mostrar
        #cv2.imshow('imagen original',img_original)
        cv2.imshow("Imagen nueva",img)
        #cv2.imshow("Imagen peque√±a", imgpequena)
        tecla=cv2.waitKey(0)
        if tecla==ord('+'):
            zoom=zoom*1.2
        elif tecla==ord('-'):
            zoom=zoom*0.8
        elif tecla==ord('o'):
            zoom=1.0
        elif tecla==ord("x"):
            break

    #cv2.waitKey(0)
    cv2.destroyWindow()

'''__main__ es el nombre interno del archivo cuando se ejecuta en si mismo'''
if __name__=="__main__":
    zoom()
